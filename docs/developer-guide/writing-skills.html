Writing Skills

Build custom skills to extend Seki's capabilities.

Skill Structure

A skill is a self-contained module with a manifest, entry point, and optional resources.

Skill Directory
skills/my-skill/
├── skill.yaml        # Manifest — metadata, triggers, permissions
├── index.ts          # Entry point — main logic
├── lib/              # Helper modules (optional)
│   └── analyzer.ts
├── models/           # ML models (optional)
│   └── sentiment.onnx
└── tests/
    └── my-skill.test.ts

Step 1: Create the Manifest

skill.yaml
name: whale-tracker
version: 1.0.0
description: Track large wallet movements and alert on significant transfers

triggers:
  events:
    - chain.transfer       # On-chain transfer events
    - chain.contract_call  # Contract interactions
  schedule:
    - cron: "*/5 * * * *"  # Every 5 minutes
  conditions:
    - event.value_usd > 100000  # Only large transfers

tools:
  required:
    - etherscan: [read_transactions, read_balances]
    - telegram: [send_message]
  optional:
    - twitter: [post_tweet]

memory:
  read: [watched_wallets, transfer_history]
  write: [transfer_history, alerts_sent]

config:
  min_value_usd:
    type: number
    default: 100000
    description: Minimum transfer value to track
  alert_channel:
    type: string
    default: "alerts"
    description: Telegram channel for notifications

Step 2: Write the Entry Point

index.ts
import { Skill, SkillContext, Event, SkillResult } from '@seki/sdk';

export default class WhaleTracker extends Skill {
  name = 'whale-tracker';

  async onEvent(ctx: SkillContext, event: Event): Promise<SkillResult> {
    const minValue = ctx.config.get('min_value_usd', 100000);

    // Skip small transfers
    if (event.data.value_usd < minValue) {
      return { action: 'skip', reason: 'below_threshold' };
    }

    // Check if wallet is already watched
    const watched = await ctx.memory.get('watched_wallets');
    const isWatched = watched.includes(event.data.from) ||
                      watched.includes(event.data.to);

    // Build alert message
    const alert = this.formatAlert(event, isWatched);

    // Send notification
    await ctx.tools.telegram.sendMessage({
      channel: ctx.config.get('alert_channel'),
      text: alert,
      parse_mode: 'HTML'
    });

    // Store in memory for pattern analysis
    await ctx.memory.append('transfer_history', {
      timestamp: event.timestamp,
      from: event.data.from,
      to: event.data.to,
      value_usd: event.data.value_usd,
      token: event.data.token,
      tx_hash: event.data.tx_hash
    });

    return {
      action: 'alert_sent',
      confidence: isWatched ? 0.9 : 0.6,
      data: { tx_hash: event.data.tx_hash }
    };
  }

  private formatAlert(event: Event, isWatched: boolean): string {
    const emoji = isWatched ? '🐋' : '🔔';
    const tag = isWatched ? '[WATCHED WALLET]' : '[NEW WHALE]';
    return `
      ${emoji} <b>${tag} Large Transfer Detected</b>
      Amount: $${event.data.value_usd.toLocaleString()}
      Token: ${event.data.token}
      From: <code>${event.data.from}</code>
      To: <code>${event.data.to}</code>
      TX: <a href="https://etherscan.io/tx/${event.data.tx_hash}">View</a>
    `.trim();
  }
}

Step 3: Lifecycle Hooks

Skills can implement lifecycle methods for initialization and cleanup:

Lifecycle Methods
export default class MySkill extends Skill {
  // Called once when skill is loaded
  async onLoad(ctx: SkillContext): Promise<void> {
    this.model = await ctx.loadModel('models/sentiment.onnx');
  }

  // Called on each matching event
  async onEvent(ctx: SkillContext, event: Event): Promise<SkillResult> {
    // Main logic
  }

  // Called on scheduled triggers
  async onSchedule(ctx: SkillContext, schedule: string): Promise<SkillResult> {
    // Periodic tasks
  }

  // Called when skill is being unloaded
  async onUnload(ctx: SkillContext): Promise<void> {
    await this.model.dispose();
  }

  // Called when evolution engine mutates parameters
  async onMutate(ctx: SkillContext, mutations: Mutation[]): Promise<void> {
    // Adapt to new parameters
  }
}

Step 5: Install & Deploy

CLI Commands
# Validate manifest and code
seki skill validate ./skills/whale-tracker

# Install locally
seki skill install ./skills/whale-tracker

# Publish to Seki skill marketplace
seki skill publish ./skills/whale-tracker

# Check status
seki skill status whale-tracker

Best Practices

  • Single Responsibility — One skill, one job. Compose complex behavior from multiple skills.
  • Declare Everything — All tools, memory keys, and events must be declared in the manifest.
  • Handle Failures — Tools can go offline. Always handle tool unavailability gracefully.
  • Return Confidence — The evolution engine uses confidence scores to optimize strategies.
  • Test with Replay — Use seki skill replay to test against historical event data.
  • Keep State in Memory — Don't use local files. Use the memory API for persistence.
Skill SDK: Full TypeScript SDK reference at @seki/sdk. Install with npm install @seki/sdk for type definitions and testing utilities.
